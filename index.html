/** * פרויקט: שרת נתונים פיננסיים מאוחד * מפתח: Expert Financial Systems Engineer * תיאור: שרת זה מקבל רשימת טיקרים, מושך נתונים היסטוריים ומאחד אותם לגיליון אחד. */ /** * פונקציית doPost המטפלת בבקשות נכנסות (JSON) */ function doPost(e) { try { // 1. קבלת הנתונים - טיפול ב-text/plain כדי למנוע בעיות CORS const requestData = JSON.parse(e.postData.contents); const { tickers, startDate, endDate, frequency } = requestData; // בדיקת תקינות בסיסית if (!tickers || !Array.isArray(tickers) || !startDate || !endDate) { throw new Error("Missing parameters: tickers (Array), startDate, or endDate."); } const freqMap = { "1d": "DAILY", "1wk": "WEEKLY", "1mo": "MONTHLY" }; const yahooFreq = frequency || "1d"; const googleFreq = freqMap[yahooFreq] || "DAILY"; // אובייקט לאחסון הנתונים הממוזגים (Key = Date, Value = {Ticker: Price}) const masterData = {}; // 2. לולאה על כל הטיקרים ומשיכת נתונים tickers.forEach(currentTicker => { let rawData = []; try { // ניסיון ראשון: Yahoo Finance עם Cookie/Crumb console.log(`מנסה למשוך את ${currentTicker} מ-Yahoo...`); rawData = fetchFromYahoo(currentTicker, startDate, endDate, yahooFreq); } catch (err) { // ניסיון שני: Fallback ל-GoogleFinance console.warn(`Yahoo נכשל עבור ${currentTicker}. עובר ל-GoogleFinance. שגיאה: ${err.message}`); rawData = fetchFromGoogleFinance(currentTicker, startDate, endDate, googleFreq); } // מיזוג הנתונים לתוך ה-masterData (Outer Join) rawData.forEach(entry => { const dKey = entry.date; if (!masterData[dKey]) masterData[dKey] = {}; masterData[dKey][currentTicker] = entry.price; }); }); // 3. בניית המערך הסופי וכתיבה לגיליון const finalSheetUrl = createFinalSpreadsheet(tickers, masterData); // 4. החזרת תגובת JSON מוצלחת return ContentService.createTextOutput(JSON.stringify({ status: "success", sheetUrl: finalSheetUrl })).setMimeType(ContentService.MimeType.JSON); } catch (error) { console.error(`שגיאה קריטית בשרת: ${error.toString()}`); return ContentService.createTextOutput(JSON.stringify({ status: "error", message: error.toString() })).setMimeType(ContentService.MimeType.JSON); } } /** * משיכת נתונים מ-Yahoo Finance באמצעות מנגנון Cookie/Crumb */ function fetchFromYahoo(ticker, startDate, endDate, interval) { const p1 = Math.floor(new Date(startDate).getTime() / 1000); const p2 = Math.floor(new Date(endDate).getTime() / 1000); const userAgent = "Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/119.0.0.0 Safari/537.36"; // א) השגת Cookie const fcRes = UrlFetchApp.fetch("https://fc.yahoo.com", { muteHttpExceptions: true, headers: { "User-Agent": userAgent } }); const cookie = fcRes.getAllHeaders()['Set-Cookie']; // ב) השגת Crumb const crumbRes = UrlFetchApp.fetch("https://query2.finance.yahoo.com/v1/test/getcrumb", { headers: { "Cookie": cookie, "User-Agent": userAgent }, muteHttpExceptions: true }); const crumb = crumbRes.getContentText(); // ג) הורדת ה-CSV const downloadUrl = `https://query1.finance.yahoo.com/v7/finance/download/${ticker}?period1=${p1}&period2=${p2}&interval=${interval}&events=history&crumb=${crumb}`; const response = UrlFetchApp.fetch(downloadUrl, { headers: { "Cookie": cookie, "User-Agent": userAgent }, muteHttpExceptions: true }); if (response.getResponseCode() !== 200) throw new Error(`Yahoo HTTP Error ${response.getResponseCode()}`); const csvData = Utilities.parseCsv(response.getContentText()); const header = csvData[0]; const dateIdx = header.indexOf("Date"); const adjCloseIdx = header.indexOf("Adj Close") !== -1 ? header.indexOf("Adj Close") : header.indexOf("Close"); return csvData.slice(1).map(row => ({ date: row[dateIdx], price: parseFloat(row[adjCloseIdx]) })).filter(item => !isNaN(item.price)); } /** * פונקציית Fallback: שימוש ב-GOOGLEFINANCE דרך גיליון זמני */ function fetchFromGoogleFinance(ticker, startDate, endDate, interval) { const tempSS = SpreadsheetApp.create(`Temp_${ticker}_${Date.now()}`); const sheet = tempSS.getSheets()[0]; // המרת תאריכים לפורמט נוסחה const start = startDate.replace(/-/g, ","); const end = endDate.replace(/-/g, ","); const formula = `=GOOGLEFINANCE("${ticker}", "price", DATE(${start}), DATE(${end}), "${interval}")`; sheet.getRange(1, 1).setFormula(formula); // המתנה לחישוב שרתים SpreadsheetApp.flush(); Utilities.sleep(2000); const values = sheet.getDataRange().getValues(); // ניקוי הקובץ הזמני DriveApp.getFileById(tempSS.getId()).setTrashed(true); // עיבוד הנתונים (דילוג על כותרות) return values.slice(1).map(row => { if (!(row[0] instanceof Date)) return null; return { date: Utilities.formatDate(row[0], Session.getScriptTimeZone(), "yyyy-MM-dd"), price: parseFloat(row[1]) }; }).filter(item => item !== null && !isNaN(item.price)); } /** * יצירת הגיליון הסופי, מיזוג הנתונים והגדרת הרשאות ציבוריות */ function createFinalSpreadsheet(tickers, masterData) { // מיון תאריכים בסדר כרונולוגי const sortedDates = Object.keys(masterData).sort((a, b) => new Date(a) - new Date(b)); // בניית מטריצת הנתונים (Header + Rows) const header = ["Date", ...tickers]; const rows = sortedDates.map(date => { const row = [date]; tickers.forEach(t => { row.push(masterData[date][t] || ""); // השארת תא ריק אם אין נתון }); return row; }); // יצירת הקובץ const ssName = `Financial Export - ${tickers.join(", ")} (${new Date().toLocaleDateString()})`; const ss = SpreadsheetApp.create(ssName); const sheet = ss.getSheets()[0]; // כתיבת הנתונים sheet.getRange(1, 1, rows.length + 1, header.length).setValues([header, ...rows]); // עיצוב בסיסי (הקפאת שורה ראשונה והדגשה) sheet.setFrozenRows(1); sheet.getRange(1, 1, 1, header.length).setFontWeight("bold").setBackground("#f3f3f3"); // 5. הגדרת הרשאות Anyone with link can View const file = DriveApp.getFileById(ss.getId()); file.setSharing(DriveApp.Access.ANYONE_WITH_LINK, DriveApp.Permission.VIEW); return ss.getUrl(); } /** * פונקציית doGet לבדיקת זמינות השרת */ function doGet() { return ContentService.createTextOutput("Finance Server is Running. Ready for POST requests.") .setMimeType(ContentService.MimeType.TEXT); }

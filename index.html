function doPost(e) { try { const data = JSON.parse(e.postData.contents); const { tickers, startDate, endDate, frequency = '1d' } = data; // המרת תאריכים לפורמט ש-Yahoo מבינה const p1 = Math.floor(new Date(startDate).getTime() / 1000); const p2 = Math.floor(new Date(endDate).getTime() / 1000); const masterData = {}; const processedTickers = []; tickers.forEach(ticker => { const csvText = fetchFromYahoo(ticker, p1, p2, frequency); if (csvText) { const rows = Utilities.parseCsv(csvText); if (rows.length > 1) { extractData(ticker, rows, masterData); processedTickers.push(ticker); } } }); if (processedTickers.length === 0) throw new Error("לא נמצאו נתונים עבור הטיקרים שנבחרו"); // בניית המערך הסופי לגיליון const finalTable = prepareTable(processedTickers, masterData); // יצירת הקובץ const ss = SpreadsheetApp.create("דוח מניות - " + processedTickers.join(", ")); const sheet = ss.getSheets()[0]; sheet.getRange(1, 1, finalTable.length, finalTable[0].length).setValues(finalTable); // שיתוף הקובץ const file = DriveApp.getFileById(ss.getId()); file.setSharing(DriveApp.Access.ANYONE_WITH_LINK, DriveApp.Permission.VIEW); return ContentService.createTextOutput(JSON.stringify({ status: "success", sheetUrl: ss.getUrl() })).setMimeType(ContentService.MimeType.JSON); } catch (err) { return ContentService.createTextOutput(JSON.stringify({ status: "error", message: err.toString() })).setMimeType(ContentService.MimeType.JSON); } } function fetchFromYahoo(ticker, p1, p2, interval) { const url = `https://query1.finance.yahoo.com/v7/finance/download/${ticker}?period1=${p1}&period2=${p2}&interval=${interval}&events=history&includeAdjustedClose=true`; const options = { muteHttpExceptions: true, headers: { "User-Agent": "Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/91.0.4472.124 Safari/537.36" } }; const response = UrlFetchApp.fetch(url, options); return response.getResponseCode() === 200 ? response.getContentText() : null; } function extractData(ticker, csvRows, masterData) { const headers = csvRows[0]; const dateIdx = headers.indexOf("Date"); // מחפש Adj Close, ואם אין (לפעמים Yahoo משנים שמות) לוקח את Close הרגיל let priceIdx = headers.indexOf("Adj Close"); if (priceIdx === -1) priceIdx = headers.indexOf("Close"); for (let i = 1; i < csvRows.length; i++) { const row = csvRows[i]; const date = row[dateIdx]; const price = row[priceIdx]; if (date && price && price !== "null") { if (!masterData[date]) masterData[date] = {}; masterData[date][ticker] = price; } } } function prepareTable(tickers, masterData) { const sortedDates = Object.keys(masterData).sort((a, b) => new Date(a) - new Date(b)); const header = ["Date", ...tickers]; const table = [header]; sortedDates.forEach(date => { const row = [date]; tickers.forEach(t => row.push(masterData[date][t] || "")); table.push(row); }); return table; }
